57  std::vector<int> wrapIndices(const std::vector<int> &indices) const
58  {
59  std::vector<int> newIndices(indices);
60  for (size_t i=0; i<dimension_; ++i) {
61  if (!GetPeriodic(i)) {
62  continue;
63  }
64 
65  int index = indices.at(i);
66  while (index < 0) { index += GetNumPoints(i); }
67  while (index >= GetNumPoints(i)) { index -= GetNumPoints(i); }

295  std::vector<int> GetIndices(const std::vector<double> &x) const
296  {
297  // Check that input vector has the correct dimensionality
298  if (x.size() != dimension_) {
299  throw std::invalid_argument("Specified point has a larger "
300  "dimensionality than the grid.");
301  }
302 
303  std::vector<int> indices(dimension_);
304  for (size_t i = 0; i < dimension_; ++i) {
305  double xpos = x.at(i);
306  if (!GetPeriodic(i)) {
307  if (xpos < GetLower(i)) {
308  indices.at(i) = -1;
309  continue;
310  } else if (xpos > GetUpper(i)) {
311  indices.at(i) = GetNumPoints(i);
312  continue;
313  }
314  }
315 
316  // To make sure, the value is rounded in the correct direction.
317  double spacing = (GetUpper(i) - GetLower(i)) / GetNumPoints(i);
318  indices.at(i) = std::floor( (xpos - GetLower(i)) / spacing);
319  }
320 
321  return wrapIndices(indices);
322  }
